  <script>
    // Sample card data
    const cards = [
      {
        type: 'hole',
        hole: 7,
        par: 4,
        yards: 438,

        tip: "Favor the left side off the tee — right is dead."
      },
      {
        type: 'weather',
        temp: 62,
        condition: 'Partly Cloudy',
        icon: '⛅',
        detail: 'Wind 8mph NW • Perfect conditions for scoring'
      },
      {
        type: 'time',
        countdown: '2h 14m',
        event: 'Tee Time',
        details: 'Pinehurst No. 2 • 2:45 PM • Hole 1'
      },
      {
        type: 'logistics',
        icon: 'bag',
        status: 'All 8 bags confirmed',
        detail: 'Delivered to bag drop at 7:42 AM'
      },
      {
        type: 'hole',
        hole: 8,
        par: 5,
        yards: 588,
        quote: "The longest hole on the course. Donald Ross designed this as a true three-shotter, but modern players can get home in two with a good drive.",
        tip: "Lay up to 100 yards if the wind is against you."
      }
    ];
    
    let currentIndex = 0;
    let isPaused = false;
    let autoAdvanceTimer;
    
    const cardWrapper = document.getElementById('cardWrapper');
    const dotsContainer = document.getElementById('dotsContainer');
    const pauseBtn = document.getElementById('pauseBtn');
    const prevBtn = document.getElementById('prevBtn');
    const nextBtn = document.getElementById('nextBtn');
    
    // Generate card HTML
    function generateCardHTML(card) {
      switch (card.type) {
        case 'hole':
          return `
            <div class="live-card">
              <div class="card-type-badge">
                <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
                  <circle cx="12" cy="12" r="10"></circle>
                  <circle cx="12" cy="12" r="3"></circle>
                </svg>
                On Course
              </div>
              <div class="card-hole-number">Hole ${card.hole}</div>
              <div class="card-hole-details">
                Par ${card.par} <span>•</span> ${card.yards} yards
              </div>
              <p class="card-quote">"${card.quote}"</p>
              <p class="card-tip">${card.tip}</p>
            </div>
          `;
        
        case 'weather':
          return `
            <div class="live-card">
              <div class="card-type-badge">
                <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
                  <path d="M17.5 19H9a7 7 0 1 1 6.71-9h1.79a4.5 4.5 0 1 1 0 9Z"></path>
                </svg>
                Weather
              </div>
              <div class="card-weather-icon">${card.icon}</div>
              <div class="card-weather-temp">${card.temp}°F</div>
              <div class="card-weather-condition">${card.condition}</div>
              <p class="card-weather-detail">${card.detail}</p>
            </div>
          `;
        
        case 'time':
          return `
            <div class="live-card">
              <div class="card-type-badge">
                <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
                  <circle cx="12" cy="12" r="10"></circle>
                  <polyline points="12 6 12 12 16 14"></polyline>
                </svg>
                Coming Up
              </div>
              <div class="card-countdown">${card.countdown}</div>
              <div class="card-event-name">${card.event}</div>
              <p class="card-event-details">${card.details}</p>
            </div>
          `;
        
        case 'logistics':
          return `
            <div class="live-card">
              <div class="card-type-badge">
                <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
                  <rect x="1" y="3" width="15" height="13"></rect>
                  <polygon points="16 8 20 8 23 11 23 16 16 16 16 8"></polygon>
                  <circle cx="5.5" cy="18.5" r="2.5"></circle>
                  <circle cx="18.5" cy="18.5" r="2.5"></circle>
                </svg>
                Logistics
              </div>
              <div class="card-logistics-icon">
                <svg width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
                  <path d="M9 11l3 3L22 4"></path>
                  <path d="M21 12v7a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2V5a2 2 0 0 1 2-2h11"></path>
                </svg>
              </div>
              <div class="card-logistics-status">${card.status}</div>
              <p class="card-logistics-detail">${card.detail}</p>
            </div>
          `;
        
        default:
          return '';
      }
    }
    
    // Generate dots
    function generateDots() {
      dotsContainer.innerHTML = cards.map((_, i) => 
        `<div class="live-dot-indicator ${i === currentIndex ? 'active' : ''}" data-index="${i}"></div>`
      ).join('');
      
      // Add click handlers
      dotsContainer.querySelectorAll('.live-dot-indicator').forEach(dot => {
        dot.addEventListener('click', () => {
          goToCard(parseInt(dot.dataset.index));
        });
      });
    }
    
    // Show card
    function showCard(index, direction = 'next') {
      const currentCard = cardWrapper.querySelector('.live-card');
      
      if (currentCard) {
        currentCard.classList.add('exiting');
        setTimeout(() => currentCard.remove(), 400);
      }
      
      setTimeout(() => {
        cardWrapper.innerHTML = generateCardHTML(cards[index]);
      }, currentCard ? 200 : 0);
      
      // Update dots
      dotsContainer.querySelectorAll('.live-dot-indicator').forEach((dot, i) => {
        dot.classList.toggle('active', i === index);
      });
    }
    
    // Navigation
    function goToCard(index) {
      currentIndex = index;
      showCard(currentIndex);
      resetAutoAdvance();
    }
    
    function nextCard() {
      currentIndex = (currentIndex + 1) % cards.length;
      showCard(currentIndex, 'next');
    }
    
    function prevCard() {
      currentIndex = (currentIndex - 1 + cards.length) % cards.length;
      showCard(currentIndex, 'prev');
    }
    
    // Auto-advance
    function startAutoAdvance() {
      autoAdvanceTimer = setInterval(() => {
        if (!isPaused) {
          nextCard();
        }
      }, 8000);
    }
    
    function resetAutoAdvance() {
      clearInterval(autoAdvanceTimer);
      if (!isPaused) {
        startAutoAdvance();
      }
    }
    
    // Toggle pause
    function togglePause() {
      isPaused = !isPaused;
      pauseBtn.innerHTML = isPaused 
        ? `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
            <polygon points="5 3 19 12 5 21 5 3"></polygon>
          </svg>`
        : `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
            <rect x="6" y="4" width="4" height="16"></rect>
            <rect x="14" y="4" width="4" height="16"></rect>
          </svg>`;
      
      if (!isPaused) {
        resetAutoAdvance();
      }
    }
    
    // Update time
    function updateTime() {
      const now = new Date();
      const timeStr = now.toLocaleTimeString('en-US', { 
        hour: 'numeric', 
        minute: '2-digit',
        hour12: true 
      });
      document.getElementById('currentTime').textContent = timeStr;
    }
    
    // Event listeners
    prevBtn.addEventListener('click', () => {
      prevCard();
      resetAutoAdvance();
    });
    
    nextBtn.addEventListener('click', () => {
      nextCard();
      resetAutoAdvance();
    });
    
    pauseBtn.addEventListener('click', togglePause);
    
    // Keyboard navigation
    document.addEventListener('keydown', (e) => {
      if (e.key === 'ArrowLeft') {
        prevCard();
        resetAutoAdvance();
      } else if (e.key === 'ArrowRight') {
        nextCard();
        resetAutoAdvance();
      } else if (e.key === ' ') {
        e.preventDefault();
        togglePause();
      }
    });
    
    // Initialize
    generateDots();
    showCard(currentIndex);
    startAutoAdvance();
    updateTime();
    setInterval(updateTime, 1000);
  </script>
