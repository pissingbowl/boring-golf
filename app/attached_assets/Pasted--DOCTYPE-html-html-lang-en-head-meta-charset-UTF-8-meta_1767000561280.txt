      card.addEventListener('dragend', function() {
        this.classList.remove('dragging');

    // Drop zones
    document.querySelectorAll('.event-list').forEach(list => {

        this.classList.add('drag-over');
        const dropZone = this.querySelector('.drop-zone');
        if (dropZone) dropZone.classList.add('active');
      });
      
      list.addEventListener('dragleave', function(e) {
        if (!this.contains(e.relatedTarget)) {
          this.classList.remove('drag-over');
          const dropZone = this.querySelector('.drop-zone');
          if (dropZone) dropZone.classList.remove('active');
        }
      });
      
      list.addEventListener('drop', function(e) {
        e.preventDefault();
        this.classList.remove('drag-over');
        const dropZone = this.querySelector('.drop-zone');
        if (dropZone) dropZone.classList.remove('active');
        
        // If dragging from pool, create new event
        if (draggedType && draggedCategory) {
          createNewEvent(this, draggedType, draggedCategory);
        }
        // If dragging existing event, move it
        else if (draggedElement && draggedElement.classList.contains('event-card')) {
          if (dropZone) {
            this.insertBefore(draggedElement, dropZone);
          }
        }
      });
    });
