    // Toggle dropdown
    function toggleVibeDropdown() {
      const switcher = document.getElementById('vibeSwitcher');
      switcher.classList.toggle('open');
    }
    
    // Set vibe
    function setVibe(vibe, element) {
      // Update body attribute
      document.body.setAttribute('data-vibe', vibe);
      
      // Update active state in dropdown
      document.querySelectorAll('.vibe-option').forEach(opt => {
        opt.classList.remove('active');
      });
      if (element) {
        element.classList.add('active');
      }
      
      // Update the button swatch
      const colors = vibeColors[vibe];
      const btnSwatch = document.getElementById('currentVibeSwatch');
      if (btnSwatch && colors) {
        const spans = btnSwatch.querySelectorAll('span');
        spans[0].style.background = colors[0];
        spans[1].style.background = colors[1];
        spans[2].style.background = colors[2];
      }
      
      // Save to localStorage
      localStorage.setItem('boring-golf-vibe', vibe);
      
      // Close dropdown
      document.getElementById('vibeSwitcher').classList.remove('open');
    }
    
    // Load saved vibe on page load
    document.addEventListener('DOMContentLoaded', () => {
      const savedVibe = localStorage.getItem('boring-golf-vibe');
      if (savedVibe && vibeColors[savedVibe]) {
        document.body.setAttribute('data-vibe', savedVibe);
        
        // Update button swatch
        const colors = vibeColors[savedVibe];
        const btnSwatch = document.getElementById('currentVibeSwatch');
        if (btnSwatch) {
          const spans = btnSwatch.querySelectorAll('span');
          spans[0].style.background = colors[0];
          spans[1].style.background = colors[1];
          spans[2].style.background = colors[2];
        }
        
        // Update active state
        document.querySelectorAll('.vibe-option').forEach(opt => {
          opt.classList.remove('active');
          if (opt.getAttribute('data-vibe') === savedVibe) {
            opt.classList.add('active');
          }
        });
      }
    });
    
    // Close dropdown when clicking outside
    document.addEventListener('click', (e) => {
      const switcher = document.getElementById('vibeSwitcher');
      if (!switcher.contains(e.target)) {
        switcher.classList.remove('open');
      }
    });
  </script>
