  <script>
    const par = 4;
    const scores = {
      marcus: 3,
      jake: 4,
      devon: 5,
      ryan: 6
    };
    
    function adjustScore(player, delta) {
      scores[player] = Math.max(1, scores[player] + delta);
      document.getElementById(`score-${player}`).textContent = scores[player];
      updateScoreStyle(player);
    }
    
    function updateScoreStyle(player) {
      const row = document.querySelector(`[data-player="${player}"]`);
      const score = scores[player];
      const diff = score - par;
      
      // Remove all classes
      row.classList.remove('birdie', 'bogey', 'double');
      
      // Find and remove existing label
      const existingLabel = row.querySelector('.score-label');
      if (existingLabel) existingLabel.remove();
      
      // Add appropriate class and label
      const inputGroup = row.querySelector('.score-input-group');
      let label = null;
      
      if (diff <= -2) {
        row.classList.add('birdie');
        label = createLabel('Eagle', 'eagle');
      } else if (diff === -1) {
        row.classList.add('birdie');
        label = createLabel('Birdie', 'birdie');
      } else if (diff === 1) {
        row.classList.add('bogey');
        label = createLabel('Bogey', 'bogey');
      } else if (diff >= 2) {
        row.classList.add('double');
        label = createLabel('Double+', 'double');
      }
      
      if (label) {
        inputGroup.appendChild(label);
      }
    }
    
    function createLabel(text, className) {
      const label = document.createElement('span');
      label.className = `score-label ${className}`;
      label.textContent = text;
      return label;
    }
    
    // Hole navigation
    let currentHole = 7;
    
    document.getElementById('prevBtn').addEventListener('click', () => {
      if (currentHole > 1) {
        currentHole--;
        updateHoleDisplay();
      }
    });
    
    document.getElementById('nextBtn').addEventListener('click', () => {
      if (currentHole < 18) {
        currentHole++;
        updateHoleDisplay();
      }
    });
    
    function updateHoleDisplay() {
      // Update hole number badge
      document.querySelector('.hole-number-badge').textContent = currentHole;
      document.querySelector('.hole-title').textContent = `Hole ${currentHole}`;
      
      // Update progress dots
      document.querySelectorAll('.hole-dot').forEach((dot, index) => {
        dot.classList.remove('current', 'completed');
        if (index + 1 < currentHole) {
          dot.classList.add('completed');
        } else if (index + 1 === currentHole) {
          dot.classList.add('current');
        }
      });
      
      // Update round status
      document.querySelector('.round-status').textContent = `Hole ${currentHole} of 18`;
    }
    
    // End round confirmation
    document.querySelector('.end-round-btn').addEventListener('click', () => {
      if (confirm('End this round? Scores will be saved.')) {
        alert('Round complete! Returning to LIVE feed...');
        // Would redirect to normal LIVE view
      }
    });
  </script>
