  <script>
    // Banner queue system
    const bannerQueue = [];
    let bannerActive = false;
    
    const sampleBanners = [
      {
        icon: '🔥',
        title: 'Skin Won — Hole 7',
        message: 'Marcus Chen wins $40 (2-hole carryover)'
      },
      {
        icon: '🎯',
        title: 'Birdie!',
        message: 'Jake Morrison birdies #8 to win the skin'
      },
      {
        icon: '💰',
        title: 'Carryover',
        message: 'Hole 9 pushes — $60 carries to #10'
      },
      {
        icon: '🏆',
        title: 'New Leader',
        message: 'Marcus Chen takes the lead with 3 skins'
      }
    ];
    
    function showBanner(bannerData) {
      const banner = document.getElementById('liveBanner');
      const icon = banner.querySelector('.banner-icon');
      const title = banner.querySelector('.banner-title');
      const message = banner.querySelector('.banner-message');
      
      icon.textContent = bannerData.icon;
      title.textContent = bannerData.title;
      message.textContent = bannerData.message;
      
      banner.classList.remove('exiting');
      banner.classList.add('visible');
      bannerActive = true;
      
      // Auto-dismiss after 4 seconds
      setTimeout(() => {
        if (bannerActive) {
          dismissBanner();
        }
      }, 4000);
    }
    
    function dismissBanner() {
      const banner = document.getElementById('liveBanner');
      banner.classList.add('exiting');
      banner.classList.remove('visible');
      bannerActive = false;
      
      // Check queue after dismiss animation
      setTimeout(() => {
        if (bannerQueue.length > 0) {
          const next = bannerQueue.shift();
          showBanner(next);
        }
      }, 500);
    }
    
    function queueBanner(bannerData) {
      if (bannerActive) {
        bannerQueue.push(bannerData);
      } else {
        showBanner(bannerData);
      }
    }
    
    // Demo: Show banners on interval
    let bannerIndex = 0;
    
    function demoBanners() {
      queueBanner(sampleBanners[bannerIndex]);
      bannerIndex = (bannerIndex + 1) % sampleBanners.length;
    }
    
    // Start demo after 2 seconds
    setTimeout(() => {
      demoBanners();
      // Then every 8 seconds
      setInterval(demoBanners, 8000);
    }, 2000);
    
    // Game tab switching
    document.querySelectorAll('.game-tab').forEach(tab => {
      tab.addEventListener('click', () => {
        document.querySelectorAll('.game-tab').forEach(t => t.classList.remove('active'));
        tab.classList.add('active');
      });
    });
  </script>
