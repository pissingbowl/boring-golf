-- Trip balances view (calculated)
CREATE OR REPLACE VIEW trip_balances AS
SELECT 
  tp.trip_id,
  tp.user_id,
  u.name as user_name,
  COALESCE(paid.total_paid, 0) as total_paid,
  COALESCE(owed.total_owed, 0) as total_owed,
  COALESCE(paid.total_paid, 0) - COALESCE(owed.total_owed, 0) as balance
FROM trip_participants tp
JOIN users u ON u.id = tp.user_id
LEFT JOIN (
  SELECT paid_by as user_id, trip_id, SUM(amount) as total_paid
  FROM expenses
  GROUP BY paid_by, trip_id
) paid ON paid.user_id = tp.user_id AND paid.trip_id = tp.trip_id
LEFT JOIN (
  SELECT es.user_id, e.trip_id, SUM(es.amount_owed) as total_owed
  FROM expense_splits es
  JOIN expenses e ON e.id = es.expense_id
  GROUP BY es.user_id, e.trip_id
) owed ON owed.user_id = tp.user_id AND owed.trip_id = tp.trip_id;

-- Trip access policy (participants only)
CREATE POLICY "Users can view trips they participate in"
  ON trips FOR SELECT
  USING (
    id IN (
      SELECT trip_id FROM trip_participants WHERE user_id = auth.uid()
    )
  );

-- Rounds access policy (trip participants)
CREATE POLICY "Trip participants can view rounds"
  ON rounds FOR SELECT
  USING (
    trip_id IN (
      SELECT trip_id FROM trip_participants WHERE user_id = auth.uid()
    )
  );

-- Scores access policy
CREATE POLICY "Trip participants can view scores"
  ON scores FOR SELECT
  USING (
    round_id IN (
      SELECT r.id FROM rounds r
      JOIN trip_participants tp ON tp.trip_id = r.trip_id
      WHERE tp.user_id = auth.uid()
    )
  );

-- Games access policy
CREATE POLICY "Trip participants can view games"
  ON games FOR SELECT
  USING (
    trip_id IN (
      SELECT trip_id FROM trip_participants WHERE user_id = auth.uid()
    )
  );

-- Expenses access policy
CREATE POLICY "Trip participants can view expenses"
  ON expenses FOR SELECT
  USING (
    trip_id IN (
      SELECT trip_id FROM trip_participants WHERE user_id = auth.uid()
    )
  );

7.1 Create Storage Bucket
In Supabase Dashboard → Storage → Create bucket:
* Name: receipts
* Public: No (private)
* File size limit: 10MB
* Allowed MIME types: image/jpeg, image/png, image/heic, application/pdf
7.2 Storage Policies
-- Allow users to upload receipts for trips they're part of
CREATE POLICY "Users can upload receipts for their trips"
ON storage.objects FOR INSERT
WITH CHECK (
  bucket_id = 'receipts' AND
  (storage.foldername(name))[1]::uuid IN (
    SELECT trip_id FROM trip_participants WHERE user_id = auth.uid()
  )
);

-- Allow users to view receipts for their trips
CREATE POLICY "Users can view receipts for their trips"
ON storage.objects FOR SELECT
USING (
  bucket_id = 'receipts' AND
  (storage.foldername(name))[1]::uuid IN (
    SELECT trip_id FROM trip_participants WHERE user_id = auth.uid()
  )
);
7.3 Upload Flow
// Client-side upload
async function uploadReceipt(tripId: string, file: File) {
  const fileName = `${tripId}/${Date.now()}-${file.name}`;
  
  const { data, error } = await supabase.storage
    .from('receipts')
    .upload(fileName, file, {
      cacheControl: '3600',
      upsert: false
    });
  
  if (error) throw error;
  
  // Get signed URL (valid 1 hour)
  const { data: urlData } = await supabase.storage
    .from('receipts')
    .createSignedUrl(fileName, 3600);
  
  return urlData.signedUrl;
}

PART 8: VENMO DEEP LINKS
8.1 Venmo URL Scheme
// Venmo deep link format
function generateVenmoLink(
  recipientUsername: string,
  amount: number,
  note: string
): string {
  const params = new URLSearchParams({
    txn: 'pay',
    recipients: recipientUsername,
    amount: amount.toFixed(2),
    note: note,
  });
  
  // Mobile deep link (opens app)
  const mobileLink = `venmo://paycharge?${params.toString()}`;
  
  // Web fallback
  const webLink = `https://venmo.com/?${params.toString()}`;
  
  return mobileLink;
}

// Leaderboard real-time updates
const subscribeToGame = (gameId: string, onUpdate: () => void) => {
  return supabase
    .channel(`game-${gameId}`)
    .on(
      'postgres_changes',
      {
        event: '*',
        schema: 'public',
        table: 'scores',
        filter: `round_id=eq.${roundId}`
      },
      onUpdate
    )
    .on(
      'postgres_changes',
      {
        event: 'INSERT',
        schema: 'public',
        table: 'game_events',
        filter: `game_id=eq.${gameId}`
      },
      (payload) => {
        // Show live banner
        showBanner(payload.new);
      }
    )
    .subscribe();
};

// Ledger real-time updates
const subscribeToLedger = (tripId: string, onUpdate: () => void) => {

    .channel(`ledger-${tripId}`)
    .on(

      },

    )
    .on(

      {
        event: '*',
        schema: 'public',
        table: 'settlements',
        filter: `trip_id=eq.${tripId}`
      },
      onUpdate
    )
    .subscribe();
};
